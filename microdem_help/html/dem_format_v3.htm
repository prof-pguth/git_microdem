MICRODEM version 3 DEM Format

The format was introduced 5/23/2010, to accommodate DEMs with more than 32,767 columns or rows.  The NumCol and NumRow changes from 2 byte integers to an unsigned 2 byte value.

tHeaderRecord = packed record   //for a DEM

  • DEMUsed : tDEMAvailable;
  • DEMPrecision : tDEMprecision;
  • DataSpacing : tSpacingUnit;
  • MaxElev, {max elevation in data set}
  • MinElev, {min elevation in data set}
  • LatInterval, {y spacing}
  • fLongInterval : shortFloat; {x spacing}
  • NumCol,NumRow : word;
  • ElevUnits : tZValueUnit;
  • DefProjName : string[7]; {Pascal, 8 bytes, byte[0] is the string length} {complex projection definition}
  • SquareID : string2; {Pascal, 3 bytes, byte[2] is the string length}
  • UTMZone : Byte; {6 degree UTM Zone number, USGS/MGRS standard: 1 = W177, 60 = E177}
  • DMAMapDefinition : tDMAMapBasicDefinition;
  • DigitizeDatum : tDigitizeDatum; {sets the datum for the DEM, and it is transformed to the desired local datum for use}
  • LatHemisph : AnsiChar; {N or S}
  • CreationInformation : string[29];
  • FIPS_Zone : SmallInt;
  • hdfSWCornerx,
  • hdfSWCornerY : Float;
  • FutureUse : array[1..15] of byte;
    •

Types:

  • tDEMAvailable = (UTMbasedDEM,ArcSecondDEM);
  • tSpacingUnit = (SpaceMeters,SpaceSeconds,SpaceMinutes,SpaceKM,Space100m, SpaceFeet,SpaceKFeet,SpaceDegrees,SpaceHundredthSecond,MercProj100m, PolarStereo100m,Space10m,SpaceTenthSecond,SpaceTenThousandthDegree,SpaceIntFeet,SpaceUSFeet);
  • tZValueUnit = (Meters,Feet,TenthMgal,Milligal,TenthGamma,Decimeters,Gammas, HundredthMGal,DeciFeet,Centimeters,OtherElev,HundredthMa,TenthPercentSlope, Undefined,zDegrees,UndefinedHundredth,lnElev,LogElev,zTenthDegrees, UndefinedThousandth,HundredthPercent,UndefinedTenth,HundredthDegree, Tens,Hundreds,Thousands,NLCD2001);
  • tDigitizeDatum = (WGS72d,WGS84d,NAD27d,NAD83d,Spherical,Local, Rectangular,LamAzEqAreaSphere,SinusEllip,PRDd,UK_OS_grid,ddDefined,MarsD,VenusD);
  • tDEMprecision = (SmallIntDEM,FloatingPointDEM,ByteDEM);
  • tDMAMapBasicDefinition = packed record
    • h_Adat,h_f : double; {added July 95}
    • h_XDat,h_YDat,h_ZDat : SmallInt; {added July 95}
    • h_DatumCode : tDatumCode; {added June 98}
    • h_EllipsCode : tEllipsCode; {added June 98}
  • end;
